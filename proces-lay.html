export function showMore() { const showMoreButton1 = document.querySelector('.brands__showmore'); const listBrends = document.querySelectorAll(".brands__slide"); const showMoreButton2 = document.querySelector('.repair-devices__showmore'); const listDiveces = document.querySelectorAll(".repair-devices__slide"); if (showMoreButton1) { showMoreButton1.addEventListener('click', () => { listBrends.forEach((el) => { if (el.closest(".show")) { el.classList.remove("show"); showMoreButton1.classList.remove("active"); showMoreButton1.textContent = "Показать все"; } else { el.classList.add("show"); showMoreButton1.classList.add("active"); showMoreButton1.textContent = "Скрыть"; } }); }); } if (showMoreButton2) { showMoreButton2.addEventListener('click', () => { listDiveces.forEach((el) => { if (el.closest(".show")) { el.classList.remove("show"); showMoreButton2.classList.remove("active"); showMoreButton2.textContent = "Показать все"; } else { el.classList.add("show"); showMoreButton2.classList.add("active"); showMoreButton2.textContent = "Скрыть"; } }); }); } } showMore()
